def discount_rate_opt(s):#s
    if ':' in s: #15020
        split=s.split(':') #
        discount_rate=(int(split[0])-int(split[1]))/int(split[0]) #,(150-20/15015020)
        return round(discount_rate,2) #
    elif s=='null':
        return np.NaN
    else:
        return float(s) #0.9
