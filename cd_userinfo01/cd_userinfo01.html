def rfm_func(x):  #x:
    level = x.apply(lambda x:'1' if x>=1 else '0') 
    label = level['R'] + level['F'] + level['M']  #100    001  111
    d = {
        '111':'',
        '011':'',
        '101':'',
        '001':'',
        '110':'',
        '010':'',
        '100':'',
        '000':''
        
    }
    result = d[label] #
    return result
